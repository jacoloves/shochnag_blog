def Nswap(n,s,top,back):
    tmp1 = top
    tmp2 = back
    top = s
    back = n
    n = tmp1
    s = tmp2
    return n, s, top, back
def Sswap(n,s,top,back):
    tmp1 = top
    tmp2 = back
    top = n
    back = s
    n = tmp2
    s = tmp1
    return n, s, top, back
    
def Eswap(e,w,top,back):

def Wswap(e,w,top,back):
